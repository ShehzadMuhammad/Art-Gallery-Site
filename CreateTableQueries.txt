CREATE TABLE artists(artistId UNSIGNED INTEGER PRIMARY KEY, name VARCHAR(30) NOT NULL, information VARCHAR(500));

CREATE TABLE artworks(artId INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, name VARCHAR(30) NOT NULL, imagePath VARCHAR(100) NOT NULL, description VARCHAR(500), artist VARCHAR(30) NOT NULL, price DECIMAL(15,2) NOT NULL);

CREATE TABLE images(imagePath VARCHAR(100) NOT NULL, size INT(6) NOT NULL, artId INT(6) AUTO_INCREMENT PRIMARY KEY);

CREATE TABLE museums(museumID UNSIGNED INT(6) AUTO_INCREMENT PRIMARY KEY, name VARCHAR(60) NOT NULL, imagePath VARCHAR(100) NOT NULL, description VARCHAR(500));

CREATE TABLE genres(name VARCHAR(50) NOT NULL);

CREATE TABLE subjects(name VARCHAR(50) NOT NULL);

CREATE TABLE shoppingcart(userID UNSIGNED INT(6) AUTO_INCREMENT PRIMARY KEY, shippingP VARCHAR(30) NOT NULL);

CREATE TABLE items(userID INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, itemName VARCHAR(30) NOT NULL);

CREATE reviews(userID INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, description VARCHAR(500) NOT NULL);
